```java
@Test
void graphProposalConfirmationOutboxAndRedisFormOneIdempotentLoop() {
    Fixture fixture = fixture();

    fixture.useDisableShortLinkProposal();

    AgentRunResult graphResult = fixture.graph.execute(fixture.batchRequest());
    AgentPendingActionView pending = fixture.singlePendingAction(graphResult);
    AgentPendingActionView executed = fixture.actionService.confirm(
            pending.actionId(),
            pending.version(),
            new AgentActionActor("owner-user", "1001", "Owner User", pending.gid()),
            "confirmed after evidence review"
    );

    assertThat(executed.status()).isEqualTo(AgentActionStatus.EXECUTED);
    assertThat(fixture.policyRepository.countByIdempotencyKey("manual:" + pending.actionId())).isEqualTo(1L);
    assertThat(fixture.outboxWorker.runNext()).isTrue();
    Object policyKeyValue = executed.result().get("policyKey");
    assertThat(policyKeyValue).isInstanceOf(String.class);
    String policyKey = (String) policyKeyValue;
    assertThat(policyKey).isNotBlank();
    assertThat(fixture.redisValue(policyKey))
            .contains("\"policyVersion\":1");
    AgentPendingActionView synced = fixture.actionService.detail(
            pending.actionId(),
            new AgentActionActor("owner-user", "1001", "Owner User", pending.gid())
    );
    assertThat(synced.result()).containsEntry("syncStatus", "SYNCED");

    AgentPendingActionView repeated = fixture.actionService.confirm(
            pending.actionId(),
            executed.version(),
            new AgentActionActor("owner-user", "1001", "Owner User", pending.gid()),
            "repeat request"
    );
    assertThat(repeated.result()).isEqualTo(synced.result());
    assertThat(fixture.policyRepository.countByIdempotencyKey("manual:" + pending.actionId())).isEqualTo(1L);
}
```
